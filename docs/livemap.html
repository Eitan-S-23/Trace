// 代码位置：LiveMap.cpp:154-166
void LiveMap::Update()
{
    if (lv_tick_elaps(priv.lastMapUpdateTime) >= CONFIG_GPS_REFR_PERIOD)
    {
        CheckPosition();          // ← 检查GPS位置变化
        SportInfoUpdate();        // 更新速度、距离、时间显示
        priv.lastMapUpdateTime = lv_tick_get();
    }
    else if (lv_tick_elaps(priv.lastContShowTime) >= 3000)
    {
        lv_obj_add_state(View.ui.zoom.cont, LV_STATE_USER_1);
    }
}
CONFIG_GPS_REFR_PERIOD：GPS更新周期，默认 30ms（在 Config.h:41 定义）

2.2 位置检查逻辑

// 代码位置：LiveMap.cpp:194-242
void LiveMap::CheckPosition()
{
    bool refreshMap = false;

    // 1. 获取GPS信息
    HAL::GPS_Info_t gpsInfo;
    Model.GetGPS_Info(&gpsInfo);

    // 2. 检查地图缩放等级是否改变
    mapLevelCurrent = lv_slider_get_value(View.ui.zoom.slider);
    if (mapLevelCurrent != Model.mapConv.GetLevel())
    {
        refreshMap = true;
        Model.mapConv.SetLevel(mapLevelCurrent);
    }

    // 3. 将GPS经纬度转换为地图坐标
    int32_t mapX, mapY;
    Model.mapConv.ConvertMapCoordinate(
        gpsInfo.longitude, gpsInfo.latitude,
        &mapX, &mapY
    );
    Model.tileConv.SetFocusPos(mapX, mapY);

    // 4. 检查地图瓦片容器是否需要更新
    if (GetIsMapTileContChanged())
        refreshMap = true;

    // 5. 如果需要刷新地图，重新加载瓦片
    if (refreshMap)
    {
        TileConv::Rect_t rect;
        Model.tileConv.GetTileContainer(&rect);
        Area_t area = { rect.x, rect.y,
                        rect.x + rect.width - 1,
                        rect.y + rect.height - 1 };
        onMapTileContRefresh(&area, mapX, mapY);
    }

    // 6. 更新地图容器位置和箭头（每次都调用，内部有缓存）
    MapTileContUpdate(mapX, mapY, gpsInfo.course);
}

3. GPS轨迹线绘制机制

3.1 轨迹系统核心组件

lv_poly_line
多段线渲染类，基于LVGL line控件，管理多条不连续线段
TrackLineFilter
轨迹线过滤器，负责区域裁剪 + START/APPEND/END/RESET事件生成
TrackPointFilter
轨迹点过滤器，点采样 + 距离过滤，减少冗余点

3.2 lv_poly_line 数据结构

class lv_poly_line
{
public:
    typedef struct
    {
        lv_obj_t* line;                  // LVGL line对象
        std::vector<lv_point_t> points; // 线段的所有点
    } single_line_t;

private:
    std::vector<single_line_t> poly_line;  // 多段线数组
    uint32_t current_index;                 // 当前活动线段索引
    lv_style_t* styleLine;                  // 线条样式
    lv_obj_t* parent;                       // 父容器
};
为什么需要多段线？ 轨迹可能跨越多个不连续区域，当轨迹移出可视区域时结束当前线段，移回时开始新线段。每个 single_line_t 代表一段连续的可见轨迹。

3.3 lv_poly_line 关键方法

// 开始新线段
void lv_poly_line::start()
{
    if (current_index >= poly_line.size()) add_line();
    single_line_t* sl = &poly_line[current_index];
    lv_obj_clear_flag(sl->line, LV_OBJ_FLAG_HIDDEN);
    lv_line_set_points(sl->line, nullptr, 0);  // 清空点
}

// 添加点到当前线段
void lv_poly_line::append(const lv_point_t* point)
{
    poly_line[current_index].points.push_back(*point);
}

// 结束当前线段，提交给LVGL
void lv_poly_line::stop()
{
    single_line_t* sl = &poly_line[current_index];
    lv_line_set_points(sl->line, get_points(sl), sl->points.size());
    current_index++;
}

// 实时追加到最后一条线段（GPS实时更新）
void lv_poly_line::append_to_end(const lv_point_t* point)
{
    single_line_t* sl = get_end_line();
    sl->points.push_back(*point);
    lv_line_set_points(sl->line, get_points(sl), sl->points.size());
}

// 重置所有线段
void lv_poly_line::reset()
{
    current_index = 0;
    for (auto& sl : poly_line) {
        lv_line_set_points(sl.line, nullptr, 0);
        sl.points.clear();
        lv_obj_add_flag(sl.line, LV_OBJ_FLAG_HIDDEN);
    }
}

3.4 TrackLineFilter 核心算法

void TrackLineFilter::PushPoint(const Point_t* point)
{
    if (GetIsPointInArea(&priv.clipArea, point))
    {
        if (!priv.inArea)
        {
            // 点刚进入区域 → 发送START_LINE事件
            const Point_t* p = priv.pointCnt > 0 ? &priv.prePoint : point;
            SendEvent(EVENT_START_LINE, p);
            priv.inArea = true;
        }
        OutputPoint(point);  // → EVENT_APPEND_POINT
    }
    else
    {
        if (priv.inArea)
        {
            // 点刚离开区域 → 发送END_LINE事件
            SendEvent(EVENT_END_LINE, point);
            priv.lineCount++;
            priv.inArea = false;
        }
    }
    priv.prePoint = *point;
    priv.pointCnt++;
}

3.5 轨迹重新加载

void LiveMap::TrackLineReload(const Area_t* area, int32_t x, int32_t y)
{
    Model.lineFilter.SetClipArea(area);
    Model.lineFilter.Reset();

    // 从DataProc加载所有历史轨迹点
    Model.TrackReload([](TrackPointFilter* filter,
                         const TrackPointFilter::Point_t* point)
    {
        LiveMap* inst = (LiveMap*)filter->userData;
        inst->Model.lineFilter.PushPoint(point->x, point->y);
    }, this);

    Model.lineFilter.PushPoint(x, y);  // 当前GPS位置
    Model.lineFilter.PushEnd();
}

3.6 轨迹事件处理

void LiveMap::onTrackLineEvent(TrackLineFilter* filter, TrackLineFilter::Event_t* event)
{
    lv_poly_line* lineTrack = instance->View.ui.track.lineTrack;
    switch (event->code)
    {
    case EVENT_START_LINE:
        lineTrack->start();
        instance->TrackLineAppend(event->point->x, event->point->y);
        break;
    case EVENT_APPEND_POINT:
        instance->TrackLineAppend(event->point->x, event->point->y);
        break;
    case EVENT_END_LINE:
        if (event->point) instance->TrackLineAppend(event->point->x, event->point->y);
        lineTrack->stop();
        break;
    case EVENT_RESET:
        lineTrack->reset();
        break;
    }
}

4. 地图瓦片路径计算

4.1 Web墨卡托投影转换

// MapConv.cpp:38-63
void MapConv::ConvertMapCoordinate(
    double longitude, double latitude,
    int32_t* mapX, int32_t* mapY)
{
    double x = (longitude + 180.0) / 360.0;
    double sinLatitude = sin(latitude * M_PI / 180.0);
    double y = 0.5 - log((1.0 + sinLatitude) / (1.0 - sinLatitude)) / (4.0 * M_PI);

    uint32_t mapSize = (uint32_t)priv.tileSize << priv.level;
    *mapX = (int32_t)(Clip(x, 0.0, 1.0) * mapSize);
    *mapY = (int32_t)(Clip(y, 0.0, 1.0) * mapSize);
}
Web墨卡托投影是大部分在线地图的标准。level=16 表示街道级别缩放。

// MapConv.cpp:99-113

// TileConv.cpp:34-73
void TileConv::SetFocusPos(int32_t x, int32_t y)
{

    const int32_t tileContWidth  = (priv.viewWidth  / tileSize + 2) * tileSize; // 512
    const int32_t tileContHeight = (priv.viewHeight / tileSize + 2) * tileSize; // 512

    priv.pointTileCont[2] = { priv.pointTileCont[0].x,                  priv.pointTileCont[0].y + tileContHeight };
    priv.pointTileCont[3] = { priv.pointTileCont[0].x + tileContWidth,  priv.pointTileCont[0].y + tileContHeight };
}

7.2 LRU缓存打开流程

// lv_img_cache.c:63-141
_lv_img_cache_entry_t* _lv_img_cache_open(const void* src, lv_color_t color, int32_t frame_id)
{
    _lv_img_cache_entry_t* cache = LV_GC_ROOT(_lv_img_cache_array);

    // 1. 老化所有缓存项（LRU）
    for (i = 0; i < entry_cnt; i++)
        if (cache[i].life > INT32_MIN + LV_IMG_CACHE_AGING)
            cache[i].life -= LV_IMG_CACHE_AGING;  // 每次减1

    // 2. 查找缓存（路径字符串匹配）
    for (i = 0; i < entry_cnt; i++) {
        if (lv_img_cache_match(src, cache[i].dec_dsc.src)) {
            // 缓存命中！增加生命值
            cached_src->life += cached_src->dec_dsc.time_to_open * LV_IMG_CACHE_LIFE_GAIN;
            return cached_src;  // ← 直接返回，不需要重新加载！
        }
    }

    // 3. 缓存未命中，替换生命值最低的项
    cached_src = &cache[0];
    for (i = 1; i < entry_cnt; i++)
        if (cache[i].life < cached_src->life) cached_src = &cache[i];

    // 4. 关闭旧图片，打开新图片
    if (cached_src->dec_dsc.src) lv_img_decoder_close(&cached_src->dec_dsc);
    uint32_t t = lv_tick_get();
    lv_img_decoder_open(&cached_src->dec_dsc, src, color, frame_id);
    cached_src->dec_dsc.time_to_open = lv_tick_elaps(t);  // 记录打开耗时

    return cached_src;
}
LRU变种策略：打开耗时越长的图片（如大图）生命值权重越高，保留越久；打开快的小图可以快速重新加载，优先被替换。

lv_res_t bin_decoder_open(lv_img_decoder_dsc_t* dsc, const void* src)
{
    lv_fs_file_t file;
    lv_fs_open(&file, src, LV_FS_MODE_RD);    // ← 调用LVGL文件系统

    uint32_t file_size;
    lv_fs_seek(&file, 0, LV_FS_SEEK_END);
    lv_fs_tell(&file, &file_size);             // = 131,072 字节
    lv_fs_seek(&file, 0, LV_FS_SEEK_SET);

    uint8_t* img_data = lv_mem_alloc(file_size);
    uint32_t bytes_read;
    lv_fs_read(&file, img_data, file_size, &bytes_read);  // ← 从SD卡读取
    lv_fs_close(&file);

    dsc->img_data = img_data;
    dsc->header.w = 256;
    dsc->header.h = 256;
    dsc->header.cf = LV_IMG_CF_TRUE_COLOR;
    return LV_RES_OK;
}

9.2 脏区域刷新机制

// lv_refr.c（伪代码）
void lv_refr_now(lv_disp_t* disp)
{
    // 1. 收集脏区域（Dirty Rectangles）
    lv_area_t dirty_areas[LV_INV_BUF_SIZE];
    uint32_t dirty_count = collect_dirty_areas(dirty_areas);

    // 2. 合并重叠区域
    merge_overlapping_areas(dirty_areas, &dirty_count);

    // 3. 对每个脏区域渲染
    for (uint32_t i = 0; i < dirty_count; i++) {
        lv_color_t* buf = draw_buf->buf_act;
        draw_objects_in_area(&dirty_areas[i], buf);

        // 4. flush回调发送到屏幕
        disp->driver->flush_cb(disp->driver, &dirty_areas[i], buf);
        while (disp->driver->flushing);  // 等待flush完成
    }
}
只重绘脏区域！ GPS位置变化 → 地图容器移动 + 轨迹线更新 → 标记脏区域 → 只重绘变化区域，不重绘整个屏幕。

11.2 SPI+DMA数据传输

// HAL_Display.cpp:191-197
void HAL::Display_SendPixels(const uint16_t* pixels, uint32_t len)
{
    digitalWrite_LOW(CONFIG_SCREEN_CS_PIN);   // 片选使能
    digitalWrite_HIGH(CONFIG_SCREEN_DC_PIN);  // 数据模式
    Display_SPI_DMA_Send(pixels, len * sizeof(uint16_t));
}

// HAL_Display.cpp:57-78
static void Display_SPI_DMA_Send(const void* buf, uint32_t size)
{
    if (size > DISP_DMA_MAX_SIZE)
    {
        // DMA最大65535字节，大数据分批传输
        Disp_DMA_TragetPoint  = (uint8_t*)buf + size;
        Disp_DMA_CurrentPoint = (uint8_t*)buf + DISP_DMA_MAX_SIZE;
        size = DISP_DMA_MAX_SIZE;
    }
    else { Disp_DMA_CurrentPoint = Disp_DMA_TragetPoint = NULL; }

    dma_channel_enable(DISP_DMA_CHANNEL, FALSE);
    DISP_DMA_CHANNEL->maddr = (uint32_t)buf;
    DISP_DMA_CHANNEL->dtcnt_bit.cnt = size;
    dma_channel_enable(DISP_DMA_CHANNEL, TRUE);  // 启动DMA
}

11.3 DMA中断处理

// HAL_Display.cpp:80-99
extern "C" void DMA1_Channel3_IRQHandler(void)
{
    if (dma_flag_get(DMA1_FDT3_FLAG) != RESET)
    {
        dma_flag_clear(DMA1_FDT3_FLAG);
        if (Disp_DMA_CurrentPoint < Disp_DMA_TragetPoint)
            // 继续传输剩余数据
            Display_SPI_DMA_Send(Disp_DMA_CurrentPoint,
                                 Disp_DMA_TragetPoint - Disp_DMA_CurrentPoint);
        else
        {
            digitalWrite_HIGH(CONFIG_SCREEN_CS_PIN);  // 片选失能
            if (Disp_Callback) Disp_Callback();       // ← 调用disp_send_finish_callback()
        }
    }
}

12. 性能优化要点

12.1 已实现的优化

1
LVGL图片缓存（LV_IMG_CACHE_DEF_SIZE = 3）
避免重复从SD卡读取相同瓦片，每次命中节省约8–10ms读取时间。代码位置：lv_conf.h:166
2
地图瓦片预加载（MapTileContPreload）
视图切换时预先加载瓦片到缓存，调用 _lv_img_cache_open() 预热。代码位置：LiveMap.cpp:306-327
3
位置缓存（避免无意义更新）
箭头位置/角度有缓存（≥1度才旋转），地图容器位置有缓存。代码位置：LiveMapView.h:64-95
4
禁用透明屏幕支持（LV_COLOR_SCREEN_TRANSP = 0）
使用快速颜色混合算法，帧率从15–20 FPS 提升至 50–60 FPS（3–4倍）。代码位置：lv_conf.h:40
5
SDIO接口（替代SPI）
SD卡读取速度从3 MB/s → 15 MB/s（5倍），是地图实时渲染的关键。代码位置：SdioCard_AT32.cpp
6
双缓冲机制
LVGL渲染和DMA传输并行，两个57KB缓冲区交替使用，大幅提升吞吐量。代码位置：lv_port_disp.cpp:36-40
7
轨迹线区域裁剪
只渲染可视区域内的轨迹点，减少 99%以上的轨迹点渲染量。代码位置：TrackLineFilter.cpp:57-81

12.2 理论性能极限

阶段时间说明
GPS更新~0.5msGPS数据读取和坐标转换
瓦片路径计算~0.1ms字符串拼接
缓存查询~0.2ms遍历3个缓存项
SD卡读取0–10ms缓存命中0ms，未命中10ms
图片解码~0msBIN格式无需解码
轨迹线更新~0.5ms追加新点到lv_poly_line
LVGL渲染5–8ms渲染地图+轨迹+箭头到缓冲区
DMA传输11ms半屏57KB数据
总计17–30ms33–59 FPS
最佳情况（缓存全命中）：约6ms，理论166 FPS，实际受 LV_DISP_DEF_REFR_PERIOD=16ms 限制为 60 FPS。
最坏情况（缓存全未命中）：4个瓦片 × 10ms = 40ms + 渲染8ms + 传输11ms = 总计 59ms ≈ 17 FPS。

14. 常见问题排查

14.1 帧率低于预期（15–20 FPS）

1
LV_COLOR_SCREEN_TRANSP = 1 — 检查 lv_conf.h:40，改为0，可提升3–4倍帧率
2
图片缓存禁用 — 检查 lv_conf.h:166 LV_IMG_CACHE_DEF_SIZE，设置为3或更大
3
频繁重绘 — 检查是否每帧都调用 lv_img_set_src()，添加路径缓存
4
SD卡使用SPI而非SDIO — 切换到SDIO接口，速度提升5倍
5
轨迹线未裁剪 — 确保调用 TrackLineFilter::SetClipArea()
